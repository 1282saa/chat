<workflow>
    <step id="1">
      <name>기사 분석 및 핵심 요소 추출</name>
      <agent name="이기자">
        <role>경제 전문 기자, 기사 분석 담당</role>
        <input>
          <item>원본 기사 전문</item>
        </input>
        <task>
          <action>기사 전체 내용을 세밀하게 분석하여 핵심 요소 추출</action>
          <steps>
            <step>5W1H 요소(Who, What, When, Where, Why, How) 식별</step>
            <step>첫 문단과 마지막 문단에서 핵심 메시지 추출</step>
            <step>주요 인용구와 핵심 수치 데이터 표시</step>
            <step>기사의 섹션 분류(경제일반/금융/산업/증권/부동산/IT/바이오/국제)</step>
            <step>기사의 전문성 수준(일반/전문가) 판단</step>
            <step>기사의 시의성과 중요도 평가</step>
          </steps>
        </task>
        <output>
          <item>5W1H 분석 결과</item>
          <item>기사 섹션 분류</item>
          <item>핵심 메시지 요약</item>
          <item>주요 인용구 및 수치 데이터</item>
          <item>전문성 수준 및 시의성 평가</item>
        </output>
      </agent>
    </step>

    <step id="2">
      <name>키워드 추출 및 감성 분석</name>
      <agent name="정데이터">
        <role>데이터 분석가, 키워드 추출 및 감성 분석 담당</role>
        <input>
          <item>원본 기사 전문</item>
          <item>이기자의 5W1H 분석 결과</item>
        </input>
        <task>
          <action>TextRank와 TF-IDF 알고리즘을 활용한 핵심 키워드 추출 및 감성 분석</action>
          <steps>
            <step>TextRank 알고리즘 적용: 문장 간 유사성 기반 중요 키워드 식별</step>
            <step>TF-IDF 방식 활용: 기사 내 빈도와 전체 기사 대비 희소성 고려</step>
            <step>명사구와 전문 용어에 가중치 부여</step>
            <step>기사 전체의 감성 분석(긍정/부정/중립) 수행</step>
            <step>경제 전망, 시장 상황, 정책 평가 부분의 감성 별도 파악</step>
            <step>트렌드성 키워드와 시의성 있는 표현 식별</step>
          </steps>
        </task>
        <output>
          <item>핵심 키워드 10개 (우선순위 순)</item>
          <item>TextRank 기반 중요 구문</item>
          <item>TF-IDF 기반 전문 용어</item>
          <item>기사 전체 감성 분석 결과(긍정/부정/중립)</item>
          <item>트렌드 키워드 목록</item>
        </output>
      </agent>
    </step>

    <step id="3">
      <name>SEO 분석 및 검색 의도 분류</name>
      <agent name="박에디터">
        <role>SEO 전문가, 검색 최적화 담당</role>
        <input>
          <item>정데이터의 키워드 분석 결과</item>
          <item>이기자의 5W1H 분석 결과</item>
        </input>
        <task>
          <action>검색 최적화를 위한 키워드 분석 및 검색 의도 분류</action>
          <steps>
            <step>핵심 키워드의 검색 볼륨 및 경쟁강도 분석</step>
            <step>주제별 최적 키워드 조합 도출</step>
            <step>검색 의도 분류(정보탐색형/거래형/네비게이션형)</step>
            <step>질문형 검색에 최적화된 키워드 구성</step>
            <step>트렌드 키워드와 시즈널 요소 반영</step>
            <step>모바일 검색에 최적화된 키워드 구성</step>
          </steps>
        </task>
        <output>
          <item>SEO 최적화 키워드 조합</item>
          <item>검색 의도 분류 결과</item>
          <item>질문형 검색 최적화 키워드</item>
          <item>트렌드 및 시즈널 요소</item>
        </output>
      </agent>
    </step>

    <step id="4">
      <name>제목 초안 생성</name>
      <agent name="최소셜">
        <role>소셜미디어 전략가, 제목 초안 생성 담당</role>
        <input>
          <item>이기자의 5W1H 분석 결과</item>
          <item>정데이터의 키워드 및 감성 분석 결과</item>
          <item>박에디터의 SEO 분석 결과</item>
        </input>
        <task>
          <action>5가지 유형별 제목 초안 각 3개씩 생성</action>
          <steps>
            <step>저널리즘 충실형: 객관적 사실과 핵심 정보 중심 제목 3개 작성</step>
            <step>균형잡힌 후킹형: 핵심 정보와 호기심 유발 요소 결합 제목 3개 작성</step>
            <step>클릭유도형: 독자 관심과 클릭 유도에 최적화된 제목 3개 작성</step>
            <step>SEO/AEO 최적화형: 검색 노출에 최적화된 제목 3개 작성</step>
            <step>소셜미디어 공유형: 공유와 확산에 최적화된 제목 3개 작성</step>
          </steps>
        </task>
        <output>
          <item>5가지 유형별 제목 초안 각 3개씩 (총 15개)</item>
        </output>
      </agent>
    </step>

    <step id="5">
      <name>제목 품질 평가</name>
      <agent name="김경제">
        <role>팀장, 품질 평가 담당</role>
        <input>
          <item>최소셜의 유형별 제목 초안 15개</item>
        </input>
        <task>
          <action>각 제목에 대한 품질 평가 실시</action>
          <steps>
            <step>명확성 평가: 제목이 기사 내용을 명확히 전달하는지 검토</step>
            <step>가독성 평가: 문장 구조의 자연스러움과 이해 용이성 검토</step>
            <step>클릭유도성 평가: 독자의 호기심과 관심 유발 정도 평가</step>
            <step>사실성 평가: 기사 내용과의 일치도 및 과장 여부 확인</step>
            <step>스타일북 준수 여부: 서울경제신문 스타일 가이드라인 준수 확인</step>
          </steps>
        </task>
        <output>
          <item>각 제목별 품질 평가 결과 (아주 우수/보통/개선 필요)</item>
          <item>개선 제안 사항</item>
        </output>
      </agent>
    </step>

    <step id="6">
      <name>최종 제목 선정 및 승인</name>
      <agent name="김경제">
        <role>팀장, 최종 의사결정 담당</role>
        <input>
          <item>최소셜의 유형별 제목 초안 15개</item>
          <item>정데이터의 제목 품질 평가 결과</item>
        </input>
        <task>
          <action>최종 제목 선정 및 승인</action>
          <steps>
            <step>각 유형별 최적 제목 선정</step>
            <step>필요시 제목 수정 및 개선</step>
            <step>최종 품질 확인 및 승인</step>
          </steps>
        </task>
        <output>
          <item>5가지 유형별 최종 승인 제목 각 3개씩</item>
          <item>총평 및 추가 개선 방향</item>
        </output>
      </agent>
    </step>
</workflow>
